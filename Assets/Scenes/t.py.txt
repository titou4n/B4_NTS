import re
from collections import Counter

# Ouvre le fichier Unity
with open("ARScene2.unity", "r", encoding="utf-8") as f:
    content = f.readlines()

# Trouve tous les identifiants du type &<nombre>
ids = re.findall(r'&(\d+)', "\n".join(content))

# Vérifie les doublons
duplicates = [item for item, count in Counter(ids).items() if count > 1]

if duplicates:
    print("Identifiants dupliqués trouvés :", duplicates)
else:
    print("Aucun identifiant dupliqué !")
